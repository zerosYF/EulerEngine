#version 330 core
layout(triangles) in;
layout(triangle_strip,max_vertices=18) out;

uniform mat4 shadowVertices[6];

out vec4 FragPos;

void main(){
	for(int face=0;face<6;face++){
		gl_Layer = face;
		for(int vtx=0;vtx<3;vtx++){
			FragPos = gl_in[vtx].gl_Position;
			gl_Position = shadowVertices[face]*FragPos;
			EmitVertex();
		}
		EndPrimitive();
	}
}
